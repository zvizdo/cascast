## THE SHARED COORDINATE CONVENTION (read first — Tasks A2, A3, A7 must match exactly)

The GLB mesh lives in a **local metric** space centered on the peak:
- **X** = east(+)/west(−) meters from `centerLng`: `x = (lng - centerLng) * metersPerDegLng`
- **Z** = south(+)/north(−) meters: `z = -(lat - centerLat) * metersPerDegLat` (three.js looks down −Z)
- **Y** = exaggerated height above the DEM floor: `y = (elev_m - minElevM) * exaggeration`

`metersPerDegLat = 111_320`; `metersPerDegLng = 111_320 * cos(centerLat_radians)` (equirectangular approx at center — fine for a single-peak bbox). `metadata.json` carries every constant so the client never re-derives them.

`metadata.json` schema (written by Task A3, consumed by A5/A7):
```json
{
  "slug": "mt-rainier",
  "bbox": { "west": -121.82, "east": -121.70, "south": 46.79, "north": 46.91 },
  "centerLat": 46.8517, "centerLng": -121.7603,
  "metersPerDegLat": 111320.0, "metersPerDegLng": 76000.0,
  "minElevM": 1500.0, "maxElevM": 4392.0,
  "exaggeration": 1.6,
  "summit": { "lat": 46.8517, "lng": -121.7603, "elevM": 4392.0 }
}
```

"""Pure DEM-array -> mesh math for the terrain bake (no network/IO)."""
from __future__ import annotations
import math
import numpy as np

METERS_PER_DEG_LAT = 111_320.0

def build_metadata(slug, dem, bbox, summit, exaggeration):
    west, south, east, north = bbox
    center_lat = (south + north) / 2
    center_lng = (west + east) / 2
    s_lng, s_lat, s_elev = summit
    return {
        "slug": slug,
        "bbox": {"west": west, "east": east, "south": south, "north": north},
        "centerLat": center_lat, "centerLng": center_lng,
        "metersPerDegLat": METERS_PER_DEG_LAT,
        "metersPerDegLng": METERS_PER_DEG_LAT * math.cos(math.radians(center_lat)),
        "minElevM": float(np.min(dem)), "maxElevM": float(np.max(dem)),
        "exaggeration": exaggeration,
        "summit": {"lng": s_lng, "lat": s_lat, "elevM": s_elev},
    }

def build_mesh(dem, bbox, minElevM, exaggeration):
    """Return (vertices Nx3 float32, faces Mx3 int32, colors Nx3 uint8).
    dem is row 0 = north edge (descending latitude), col 0 = west edge."""
    rows, cols = dem.shape
    west, south, east, north = bbox
    center_lat = (south + north) / 2
    center_lng = (west + east) / 2
    mpd_lat = METERS_PER_DEG_LAT
    mpd_lng = METERS_PER_DEG_LAT * math.cos(math.radians(center_lat))
    # lon per col (west->east), lat per row (north->south)
    lons = np.linspace(west, east, cols)
    lats = np.linspace(north, south, rows)
    xs = (lons - center_lng) * mpd_lng                      # (cols,)
    zs = -(lats - center_lat) * mpd_lat                     # (rows,)
    gx, gz = np.meshgrid(xs, zs)                            # (rows, cols)
    gy = (dem - minElevM) * exaggeration
    verts = np.column_stack([gx.ravel(), gy.ravel(), gz.ravel()]).astype("float32")
    # faces: two triangles per cell
    idx = np.arange(rows * cols).reshape(rows, cols)
    tl, tr, bl, br = idx[:-1, :-1], idx[:-1, 1:], idx[1:, :-1], idx[1:, 1:]
    faces = np.concatenate([
        np.stack([tl, bl, br], axis=-1).reshape(-1, 3),
        np.stack([tl, br, tr], axis=-1).reshape(-1, 3),
    ]).astype("int32")
    colors = _hypsometric_hillshade(dem, exaggeration)
    return verts, faces, colors

def _hypsometric_hillshade(dem, exaggeration):
    """Muted hypsometric tint modulated by a simple hillshade (Nx3 uint8)."""
    rng = max(float(np.ptp(dem)), 1.0)
    t = (dem - float(np.min(dem))) / rng                    # 0..1 by elevation
    # calm slate->stone ramp (low = deep slate, high = pale stone)
    low = np.array([60, 72, 84]); high = np.array([214, 210, 200])
    tint = (low + (high - low) * t[..., None]).astype("float32")
    # hillshade from gradient (sun NW, alt 45deg)
    gy, gx = np.gradient(dem * exaggeration)
    slope = np.pi / 2 - np.arctan(np.hypot(gx, gy))
    aspect = np.arctan2(-gx, gy)
    az, alt = math.radians(315), math.radians(45)
    hs = (np.sin(alt) * np.sin(slope) +
          np.cos(alt) * np.cos(slope) * np.cos(az - aspect))
    hs = np.clip(0.6 + 0.4 * hs, 0, 1)                      # keep it gentle
    shaded = np.clip(tint * hs[..., None], 0, 255).astype("uint8")
    return shaded.reshape(-1, 3)
```

"""One-shot terrain bake CLI. Run locally: python -m functions.tools.build_terrain --mountain mt-rainier
(from repo root, with functions/.venv active + requirements-terrain.txt installed)."""
from __future__ import annotations
import argparse, io, json
import numpy as np
import httpx
import rasterio
import trimesh
from functions.tools import terrain_mesh as tm
from functions.shared import storage_client as sc

DEM_URL = "https://elevation.nationalmap.gov/arcgis/rest/services/3DEPElevation/ImageServer/exportImage"

def fetch_dem(lat: float, lng: float, span: float, size: int = 256):
    """Return (dem float32 [rows,cols], bbox (w,s,e,n)). Row 0 = north."""
    bbox = (lng - span, lat - span, lng + span, lat + span)  # w,s,e,n
    params = {
        "bbox": f"{bbox[0]},{bbox[1]},{bbox[2]},{bbox[3]}",
        "bboxSR": 4326, "imageSR": 4326, "size": f"{size},{size}",
        "format": "tiff", "pixelType": "F32",
        "interpolation": "RSP_BilinearInterpolation", "f": "image",
    }
    r = httpx.get(DEM_URL, params=params, timeout=120)
    r.raise_for_status()
    with rasterio.open(io.BytesIO(r.content)) as ds:
        dem = ds.read(1).astype("float32")
    dem = np.where(dem < -1e5, np.nan, dem)         # nodata guard
    dem = np.nan_to_num(dem, nan=float(np.nanmin(dem)))
    return dem, bbox

def bake_one(slug, lat, lng, summit_elev_m, span=0.06, exaggeration=1.6):
    dem, bbox = fetch_dem(lat, lng, span)
    meta = tm.build_metadata(slug, dem, bbox, summit=(lng, lat, summit_elev_m), exaggeration=exaggeration)
    verts, faces, colors = tm.build_mesh(dem, bbox, minElevM=meta["minElevM"], exaggeration=exaggeration)
    mesh = trimesh.Trimesh(vertices=verts, faces=faces, vertex_colors=colors, process=False)
    glb = trimesh.exchange.gltf.export_glb(mesh)
    sc.write_terrain_model(slug, glb)
    sc.write_terrain_meta(slug, json.dumps(meta))
    print(f"baked {slug}: {len(verts)} verts, {len(glb)} GLB bytes")

# ft->m for summit elevations from mountains-data (feet)
FT_TO_M = 0.3048

def main():
    from scripts_mountains import MOUNTAINS  # see Step 6 for the loader
    ap = argparse.ArgumentParser()
    ap.add_argument("--mountain"); ap.add_argument("--all", action="store_true")
    ap.add_argument("--span", type=float, default=0.06); ap.add_argument("--exaggeration", type=float, default=1.6)
    a = ap.parse_args()
    targets = MOUNTAINS if a.all else [m for m in MOUNTAINS if m["slug"] == a.mountain]
    for m in targets:
        bake_one(m["slug"], m["lat"], m["lng"], m["summit"] * FT_TO_M, a.span, a.exaggeration)

if __name__ == "__main__":
    main()
```

export async function readTerrainModel(
  mountainId: string,
): Promise<{ buffer: Buffer; contentType: string } | null> {
  const f = getStorage().bucket(requireEnv("GCS_BUCKET_TERRAIN")).file(`${mountainId}/terrain.glb`);
  const [present] = await f.exists();
  if (!present) return null;
  const [contents] = await f.download();
  return { buffer: contents, contentType: "model/gltf-binary" };
}

export async function readTerrainMeta(mountainId: string): Promise<unknown | null> {
  const f = getStorage().bucket(requireEnv("GCS_BUCKET_TERRAIN")).file(`${mountainId}/metadata.json`);
  const [present] = await f.exists();
  if (!present) return null;
  const [contents] = await f.download();
  return JSON.parse(contents.toString("utf8"));
}

```ts
export interface TerrainMeta {
  slug: string;
  bbox: { west: number; east: number; south: number; north: number };
  centerLat: number; centerLng: number;
  metersPerDegLat: number; metersPerDegLng: number;
  minElevM: number; maxElevM: number; exaggeration: number;
  summit: { lng: number; lat: number; elevM: number };
}
export interface RouteLine { name: string; grade?: string; trailhead?: string; source?: string; illustrative?: boolean; points: [number, number, number][] }

export const ftToM = (ft: number) => ft * 0.3048;
export const elevToMeshY = (m: TerrainMeta, elevM: number) => (elevM - m.minElevM) * m.exaggeration;

export function llaToMesh(m: TerrainMeta, lng: number, lat: number, elevM: number): [number, number, number] {
  const x = (lng - m.centerLng) * m.metersPerDegLng;
  const z = -(lat - m.centerLat) * m.metersPerDegLat;
  const y = elevToMeshY(m, elevM);
  return [x, y, z];
}

export function parseRoutes(fc: unknown): RouteLine[] {
  const features = (fc as { features?: unknown[] })?.features ?? [];
  const out: RouteLine[] = [];
  for (const f of features as { properties?: Record<string, unknown>; geometry?: { type?: string; coordinates?: unknown } }[]) {
    if (f?.geometry?.type !== "LineString") continue;
    const coords = f.geometry.coordinates as [number, number, number][];
    const p = f.properties ?? {};
    out.push({
      name: String(p.name ?? "Route"),
      grade: p.grade as string | undefined,
      trailhead: p.trailhead as string | undefined,
      source: p.source as string | undefined,
      illustrative: p.illustrative as boolean | undefined,
      points: coords.map((c) => [c[0], c[1], c[2] ?? 0]),
    });
  }
  return out;
}
```
